function updateBarChart() { const {data: disp, title} = getDisplayData(); const barData = sourceKeys .map(k => ({source: k, value: disp[k] || 0})) .filter(d => d.value > 0) .sort((a,b) => b.value - a.value) .slice(0,15); // Increased left margin to give more room for long wrapped labels const margin = {top: 40, right: 30, bottom: 20, left: 260}; // ← was 220, now 260 const w = 520 - margin.left - margin.right; const h = 380 - margin.top - margin.bottom; const svg = d3.select("#bar-chart") .html("") .append("svg") .attr("width", w + margin.left + margin.right) .attr("height", h + margin.top + margin.bottom) .append("g") .attr("transform", `translate(${margin.left},${margin.top})`); const x = d3.scaleLinear() .domain([0, d3.max(barData, d => d.value) * 1.1]) .range([0, w]); const y = d3.scaleBand() .domain(barData.map(d => d.source)) .range([0, h]) .padding(0.32); // ← was 0.25, now more vertical spacing // Bars svg.selectAll("rect") .data(barData) .join("rect") .attr("x", 0) .attr("y", d => y(d.source)) .attr("width", d => x(d.value)) .attr("height", y.bandwidth()) .attr("fill", d => sourceColor(d.source)) .on("mouseover", (event,d) => { tooltip.style("visibility","visible") .html(`${d.source}
${d.value.toFixed(2)} Mtoe`); }) .on("mousemove", event => { tooltip.style("top",(event.pageY-15)+"px") .style("left",(event.pageX+15)+"px"); }) .on("mouseout", () => tooltip.style("visibility","hidden")); // Y-axis with text wrapping const yAxisGroup = svg.append("g") .attr("class", "y-axis") .call(d3.axisLeft(y).tickSize(0).tickPadding(6)); yAxisGroup.selectAll("text") .style("text-anchor", "end") .each(function() { const text = d3.select(this); const words = text.text().split(/\s+/).reverse(); let word; let line = []; let lineNumber = 0; const lineHeight = 1.35; // ems const yPos = text.attr("y"); const dy = parseFloat(text.attr("dy")) || 0; text.text(null); while (word = words.pop()) { line.push(word); text.text(line.join(" ")); if (this.getComputedTextLength() > 230) { // max width before wrap (~260 - some padding) line.pop(); text.text(line.join(" ")); line = [word]; lineNumber++; text.append("tspan") .attr("x", -6) .attr("y", yPos) .attr("dy", `${lineNumber * lineHeight}em`) .text(word); } } }); // X-axis svg.append("g") .attr("transform", `translate(0,${h})`) .call(d3.axisBottom(x)); svg.append("text") .attr("x", w/2) .attr("y", -10) .attr("text-anchor","middle") .style("font-weight","bold") .text(`Top Sources – ${title} (Total: ${disp.Total.toFixed(2)} Mtoe)`); }
